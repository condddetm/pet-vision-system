\section{模型选型与方法}
% =============================================================================

\begin{frame}{方法概览：三大模型组合}
\begin{center}\footnotesize
\begin{tabular}{L{2.3cm}L{4.4cm}L{5cm}}
  \toprule
  \textbf{\color{NWUBlue}任务} & \textbf{\color{NWUBlue}模型} & \textbf{\color{NWUBlue}设计要点} \\
  \midrule
  图像分类 & ResNet-34（ImageNet 预训练）& 残差连接 + 迁移学习；\texttt{fc} 层替换为 Linear(512, 37) \\
  分类基线1 & ResNet-34（随机初始化）& 对照组：验证迁移学习收益 \\
  分类基线2 & SimpleCNN（自定义 4 层）& 容量对照：参数量仅 0.37M（1/50 ResNet-34）\\
  语义分割 & U-Net（ResNet-34 编码器）& 编码器共享主干 + 跳跃连接 \\
  可解释性 & Grad-CAM \texttt{@ layer4[-1]} & 类激活热力图，JET 调色板叠加原图 \\
  \bottomrule
\end{tabular}
\end{center}

\vspace{0.6em}
\begin{columns}[T]
  \begin{column}{0.48\linewidth}
    \begin{block}{\footnotesize\faBrain\ 分类核心：迁移学习}
      \scriptsize 保留 ImageNet 骨干权重，仅替换最后 FC 层；Adam 优化器、$\eta = 10^{-3}$、100 epoch。
    \end{block}
  \end{column}
  \begin{column}{0.48\linewidth}
    \begin{block}{\footnotesize\faPuzzlePiece\ 分割核心：主干共享}
      \scriptsize U-Net 编码器=ResNet-34（与分类一致）；BCEWithLogits 损失；30 epoch 训练至 Dice 收敛。
    \end{block}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{Grad-CAM 可解释性流程}
\begin{center}
\begin{tikzpicture}[
  node distance=0.4cm,
  every node/.style={font=\tiny, transform shape},
  pstep/.style={rectangle, rounded corners=3pt, draw=NWUBlue, very thick,
                top color=NWUBlue!5, bottom color=NWUBlue!25, minimum width=2.2cm, minimum height=0.75cm, align=center},
  data/.style={rectangle, rounded corners=3pt, draw=teal!60!black, very thick,
               top color=teal!5, bottom color=teal!25, minimum width=2.2cm, minimum height=0.75cm, align=center},
  pout/.style={rectangle, rounded corners=3pt, draw=AccentOrange!80!black, very thick,
               top color=AccentOrange!15, bottom color=AccentOrange!45, minimum width=2.2cm, minimum height=0.75cm, align=center},
  arrow/.style={-{Stealth[length=2mm]}, semithick, draw=gray!60!black}
]
  \node[data] (input) {输入图像\\$224\!\times\!224$};
  \node[pstep, right=of input] (fwd) {前向传播\\ResNet-34};
  \node[pstep, right=of fwd] (logit) {Top-1 logit\\$y_{c^*}$};
  \node[pstep, right=of logit] (back) {反向至\\\texttt{layer4[-1]}};
  \node[pstep, below=1.1cm of back] (grad) {通道梯度};
  \node[pstep, left=of grad] (gap) {GAP → $\alpha_k$};
  \node[pstep, left=of gap] (wt) {$\sum_k \alpha_k A^k$};
  \node[pout, left=of wt] (cam) {ReLU + JET\\叠加原图};
  \draw[arrow] (input) -- (fwd); \draw[arrow] (fwd) -- (logit); \draw[arrow] (logit) -- (back);
  \draw[arrow] (back.south) -- (grad.north);
  \draw[arrow] (grad) -- (gap); \draw[arrow] (gap) -- (wt); \draw[arrow] (wt) -- (cam);
\end{tikzpicture}
\end{center}

\vspace{0.2em}
\begin{columns}[T]
  \begin{column}{0.5\linewidth}
    \begin{block}{\footnotesize 权重公式}
    \footnotesize\vspace{-0.3em}
    \[ \alpha_k^{c^*} = \frac{1}{HW}\sum_{h,w}\frac{\partial y_{c^*}}{\partial A_{h,w}^k} \]
    \[ L^{c^*} = \mathrm{ReLU}\left(\sum_k \alpha_k^{c^*} A^k\right) \]
    \end{block}
  \end{column}
  \begin{column}{0.48\linewidth}
    \begin{itemize}\footnotesize
      \item \textbf{红色区域}：判别关键依据
      \item \textbf{蓝色区域}：无关/背景
      \item 上采样至原图尺寸，按 0.55 : 0.45 混合
      \item 验证模型是否关注\textbf{语义正确}的特征
    \end{itemize}
  \end{column}
\end{columns}
\end{frame}
